def clean_data(input_file, output_file):
    seen_data = set()

    with open(input_file, 'r') as input_file:
        lines = input_file.readlines()

    cleaned_lines = []
    current_entry = ""

    for line in lines:
        if line.startswith("SSID:"):
            current_entry = line
        else:
            current_entry += line
            if line == "-------------------------------\n" and current_entry not in seen_data:
                seen_data.add(current_entry)
                cleaned_lines.append(current_entry)

    with open(output_file, 'w') as output_file:
        output_file.writelines(cleaned_lines)

input_file_path = "your_input_file.txt"
output_file_path = "cleaned_data.txt"
clean_data(input_file_path, output_file_path)
